--INFORMACION SOBRE LAS TABLAS DE NUESTA BBDD--

SELECT TABLE_NAME FROM USER_TABLES;

-- INFORMACION SOBRE LAS COLUMNAS DE UNA TABLA--

SELECT COLUMN_NAME,DATA_TYPE,DATA_DEFAULT,DATA_PRECISION,DATA_SCALE,NULLABLE
FROM USER_TAB_COLUMNS
WHERE TABLE_NAME='TIENDAS';

--INFORMACION SOBRE LAS RESTRICCIONES DE UNA TABLA-- 

SELECT CC.CONSTRAINT_NAME,C.CONSTRAINT_TYPE,CC.TABLE_NAME,CC.COLUMN_NAME,C.DELETE_RULE
FROM USER_CONS_COLUMNS CC, USER_CONSTRAINTS C
WHERE CC.CONSTRAINT_NAME = C.CONSTRAINT_NAME 
AND CC.TABLE_NAME IN('EMPLE','DEPART');
